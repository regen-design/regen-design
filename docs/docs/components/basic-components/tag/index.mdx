# 标签 Tag

通常用来展示一些属性，偶尔也用来当一些备选的属性。

## 演示

<BrowserOnly>
  {() => (<ThemeProvider>
    <Grid xGap={16} cols={2} style={{
      alignItems: 'flex-start'
    }}>
      <Grid yGap={16} cols={1}>
        <CodeBlock title="基础"
                   description={'它有不同的类型。'}
                   code={String(BasicCode)}>
          <Basic />
        </CodeBlock>
        <CodeBlock title="尺寸"
                   description={'它有不同的尺寸。'}
                   code={String(SizeCode)}>
          <Size />
        </CodeBlock>
        <CodeBlock title="禁用"
                   code={String(DisabledCode)}>
          <Disabled />
        </CodeBlock>
      </Grid>
      <Grid yGap={16} cols={1}>
        <CodeBlock title="可关闭的"
                   code={String(CloseableCode)}>
          <Closeable />
        </CodeBlock>
        <CodeBlock title="无边框"
                   code={String(NoBorderedCode)}>
          <NoBordered />
        </CodeBlock>
      </Grid>
    </Grid>
  </ThemeProvider>)}
</BrowserOnly>
